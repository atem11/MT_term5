grammar testGrammar

@import {
import Lab2.parserLab2.Tree;
}

s   : nre nend
    ;

nend: Eps
    | s_
    ;

s_  : OR nre nend
    ;

nre : os next
    ;

next: Eps
    | nre_
    ;

nre_: os next
    ;

os  : re klini
    ;

klini:Z
    | Q
    | Eps
    ;

re  : LET
    | BRO s BRC
    ;

BRO : "\\(" ;
BRC : "\\)" ;
OR  : "\\|" ;
Q   : "\\?" ;
Z   : "\\*" ;
LET : "[a-z]+" ;
